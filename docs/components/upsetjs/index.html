Addons#

Live Editor
// import React from 'react';
function Addons() {
  const SetAddon = ({ width, height }) => {
    return <rect x={1} y={1} width={width - 2} height={height - 2} fill="red"></rect>;
  };
  const SetAddon2 = ({ width, height }) => {
    return <rect x={1} y={1} width={width - 2} height={height - 2} fill="green"></rect>;
  };
  const [selection, setSelection] = React.useState(null);
  return (
    <UpSetJS
      width={780}
      height={500}
      sets={GameOfThronesDataSet.sets}
      selection={selection}
      onHover={setSelection}
      setAddons={[
        {
          name: 'red',
          size: 40,
          render: SetAddon,
        },
        {
          name: 'green',
          position: 'before',
          size: 20,
          render: SetAddon2,
        },
      ]}
      combinationAddons={[
        {
          name: 'red',
          size: 40,
          render: SetAddon,
        },
        {
          name: 'green',
          position: 'before',
          size: 20,
          render: SetAddon2,
        },
      ]}
    />
  );
}
