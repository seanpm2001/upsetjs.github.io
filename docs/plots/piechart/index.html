// import React from 'react';
function Example() {
  const [selection, setSelection] = useState(null);
  const elems = useMemo(
    () =>
      Array(100)
        .fill(0)
        .map(() => ({
          a: Math.floor(Math.random() * 4).toString(),
        })),
    []
  );
  return (
    <PieChart selection={selection} onHover={setSelection} width={500} height={500} elems={elems} attr="a" title="As" />
  );
}
